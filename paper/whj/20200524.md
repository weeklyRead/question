- Q8 : 以下实例分别输出什么？
    - example1:
    ```js
        for(var i = 0; i < 5; i++){
            setTimeout(function(){
                console.log(i)
            }, 1000)
        }
    ```
    - example2:
    ```js
        for(var i = 0; i < 5; i++){
            (function(j){
                setTimeout(function(){
                    console.log(j)
                }, j*1000)
            })(i)
        }
    ```
    - example3:
    ```js
        for(let i = 0; i < 5; i++){
            setTimeout(function(){
                console.log(i)
            },1000)
        }
    ```
